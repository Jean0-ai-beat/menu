        function filterPlayers(role) {

            });

            

            const activeBtn = document.getElementById(`btn-${role}`);

            activeBtn.className = "filter-btn px-5 py-2.5 rounded-xl text-xs font-bold uppercase bg-red-600 text-white transition";


            renderRoster(role);

        }
